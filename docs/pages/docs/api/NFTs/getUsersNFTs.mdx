This API returns a list of NFTs that users own. You can also filter by the contract address of the NFT.

### API operation

```jsx
lensjs
  .GetUsersNFTs({
    ownerAddress: '0x54be3a794282c030b15e43ae2bb182e14c409c5e',
    limit: 10,
    chainIds: [1],
  })
  .then((res) => {
    console.log(res);
  })
  .catch((err) => {
    console.log(err);
  });
```

### Full Code Example

```jsx
import lensjs from 'lensjs';
import { useEffect } from 'react';

export default function Example({ ownerAddress, contractAddress, chainIds, limit }) {
  useEffect(() => {
    lensjs
      .GetUsersNFTs({
        ownerAddress,
        contractAddress,
        chainIds,
        limit,
      })
      .then((res) => {
        console.log(res);
      })
      .catch((err) => {
        console.log(err);
      });
  }, []);
}
```
