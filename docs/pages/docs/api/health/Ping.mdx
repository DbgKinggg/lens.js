Ping API `ping()` returns if the API is online.

### API operation

```jsx
lensjs
  .ping()
  .then((res) => {
    console.log(res);
  })
  .catch((err) => {
    console.log(err);
  });
```

### API response

```json
{
  "ping": "pong"
}
```

### Full Code Example

```jsx
import lensjs from 'lensjs';
import { useEffect } from 'react';

export default function Example() {
  useEffect(() => {
    lensjs
      .ping()
      .then((res) => {
        console.log(res);
      })
      .catch((err) => {
        console.log(err);
      });
  }, []);
}
```
