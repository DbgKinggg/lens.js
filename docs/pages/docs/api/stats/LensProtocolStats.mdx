Lens Protocol Stats API `stats()` returns the global protocol stats.

### API operation

```jsx
lensjs
  .GetGlobalProtocolStats()
  .then((res) => {
    console.log(res);
  })
  .catch((err) => {
    console.log(err);
  });
```

### Full Code Example

```jsx
import lensjs from 'lensjs';
import { useEffect } from 'react';

export default function Example() {
  useEffect(() => {
    lensjs
      .GetGlobalProtocolStats()
      .then((res) => {
        console.log(res);
      })
      .catch((err) => {
        console.log(err);
      });
  }, []);
}
```
